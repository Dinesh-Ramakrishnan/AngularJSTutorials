Default syntax: {{ expression | [ filter_name ]} }
{{13.5 | currency}}

Currency filter syntax: {{ currency_expression | currency[:symbol] }}
{{5.5 | currency:"INR "}}

date filter syntax: {{ UTC date value | date[:format }}
{{1330093479000 | date}}
{{1330093479000 | date:"medium"}}
{{1330093479000 | date:"yyyy/MM"}}

'filter' filter syntax: {{ array | json }}
{{texts|filter:'h'}}
{{numbers|filter:'3'}}
{{persons|filter:{fname:'t'} }}
{{persons|filter:{$:'r'} }}
{{persons|filter:{fname:'o', lname:'o'} }}

json filter syntax: {{ json text | json }}
{{ {"car": "Audi", "make": "4X"} }}
{{ {"car": "Audi", "make": "4X"} | json}}

limitTo filter syntax: {{ array | limitTo[:count] }}
{{numbers}}
{{numbers|limitTo:4}}

Case change filter syntax: {{ string | [lowercase/uppercase] }}
{{ "All My TeXt." | lowercase}}
{{ "acronym" | uppercase }}

number filter syntax: {{ numeric value | number:[decimal places] }}
{{ "3.14159" | number}}
{{ "3.14159" | number:0}}
{{ "3.14159" | number:2}}
{{ "314159.12345" | number}}

orderBy filter syntax: {{ array | orderBy:"[-]fieldName" }}; '-' puts in reverse order.
{{persons |orderBy:"-fname" }}
{{persons |orderBy:"lname" }}
